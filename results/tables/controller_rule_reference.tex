\texttt{Phase initialization / reset} &
\texttt{phase\_initialization} &
initialization &
New active phase uses that phase configured reward component initial weights. &
No metric trigger. Initializes weights from PhaseConfig.reward\_components. \\
\midrule
\texttt{Parseable Guard Fired} &
\texttt{parseable\_guard} &
guard update &
parseable\_answer\_rate < 0.85. &
Sets parseable\_reward = max(current, 0.75). \\
\midrule
\texttt{Format Guard Fired} &
\texttt{format\_guard} &
guard update &
solution\_tag\_compliance < 0.9 OR reasoning\_tag\_compliance < 0.88 OR malformed\_answer\_rate > 0.1. &
Sets format\_reward = max(current, 0.75). \\
\midrule
\texttt{Finish Guard Fired} &
\texttt{finish\_guard} &
guard update &
truncation\_rate > 0.15 OR average\_completion\_tokens > 0.85 * max\_completion\_length. &
Adds +0.25 to finished\_reward before bounds/floors are applied. \\
\midrule
\texttt{Stable Structure} &
\texttt{stable\_structure} &
helper condition &
parseable >= 0.92 AND solution >= 0.95 AND reasoning >= 0.93 AND malformed <= 0.05 AND truncation <= 0.08. &
Does not change weights by itself; required for correctness escalation. \\
\midrule
\texttt{Stable Window Ready} &
\texttt{stable\_window\_ready} &
helper condition &
At least 2 prior/current checkpoint metric entries exist in phase history. &
Enables exact-match plateau check. \\
\midrule
\texttt{Correctness Plateau} &
\texttt{correctness\_plateau} &
helper condition &
current normalized\_exact\_match minus previous window exact match < 0.02. &
Does not change weights by itself; required for correctness escalation. \\
\midrule
\texttt{Correctness Rule Fired} &
\texttt{correctness\_escalation} &
update rule &
stable\_structure AND stable\_window\_ready AND correctness\_plateau. &
Adds +0.5 to correctness\_reward before bounds/floors are applied. \\
\midrule
\texttt{Post-update Floors} &
\texttt{component\_floors} &
post-update constraint &
Always applied after rule updates. &
Ensures format\_reward >= 0.25, parseable\_reward >= 0.25, finished\_reward >= 0.5. \\
\midrule
\texttt{Component Bounds Clamp} &
\texttt{clamped\_components} &
post-update constraint record &
Any reward weight falls outside its RewardComponentConfig min/max bound. &
Clamps to component min/max and records the changed component. \\
\bottomrule
\end{longtable}
\vspace{0.35em}
\noindent\textbf{Notes.} Helper conditions are recorded in controller decisions but do not directly update reward weights. Guard and escalation updates are followed by fixed component floors and component min/max clamping.
